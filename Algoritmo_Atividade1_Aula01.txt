Exercicio 2:


algoritmo calculo_do_imc
real peso
real altura
real imc
inicio
	escreva "Digite o seu peso: "
	leia peso
	escreva "Digite a sua altura: "
	leia altura
	media (peso/(altura*altura))
	imprima "O seu IMC é: " + imc
fim

algoritmo volume_do_cilindro

real altura
real volume
inicio
